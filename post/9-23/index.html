升序链表转换成降序链表

采用的是头插法, ~翻转~

#include <stdio.h>
#include <stdlib.h>
#define null 0

typedef struct LNode{
    int data;
    struct LNode *next;
} LNode, * Linklist;

// A和B是提供了两个带头结点的升序链表，转换成一个降序链表C
int reverse_link (Linklist &A, Linklist &B)
{
    Linklist C = (Linklist)malloc(sizeof(LNode));
    C->next = null;
    Linklist pa = A->next;
    Linklist pb = B->next;
    while (pa && pb) {
        if (pa->data <= pb->data) {
            pa->next = C->next;
            C->next = pa;
            pa = A->next;
        } else {
            pb->next = C->next;
            C->next = pb;
            pb = B->next;
        }
    }

    while (pa){ //A表非空，把剩余的元素头插入到C表中
        pa->next = C->next;
        C->next = pa;
        pa = A->next;
    }

    while (pb){ //B表非空，把剩余的元素头插入到C表中
        pb->next = C->next;
        C->next = pb;
        pb= B->next;
    }
}
